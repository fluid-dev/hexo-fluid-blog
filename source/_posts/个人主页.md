[蓝奏云优享版](https://www.ilanzou.com/s/e9iN7Pa2)

配置文件格式:

```json
{
  "name_zh": "中文名",
  "name_en": "英文名",
  "theme_color": "主题色(带#)",
  "slogans": {
    "under_name": "名下Slogan",
    "under_avatar": "头像下Slogan",
    "bottom_left": "底下左边Slogan",
    "bottom_right": "底下右边Slogan"
  },
  "avatar_path": "头像(支持网络URL)",
  "articles": [ // 文章
    {
      "title": "标题",
      "desc": "介绍",
      "tag": [
        "标签"
      ],
      "markdown": "文章MD文件(支持网络URL)"
    }
  ],
  "projects": [ // 项目
    {
      "title": "名字",
      "description": "介绍",
      "cover": "封面(可选 支持网络URL)",
      "url": "点击后跳转到的URL"
    }
  ],
  "about": { // 关于
    "description": "个人介绍",
    "urls": [ // 相关链接
      {
        "name": "名称",
        "url": "URL",
        "icon": "图标(可选 支持网络URL)"
      }
    ]
  }
}
```
